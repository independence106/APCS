public static boolean arePermutations(int[] arr1, int[] arr2) {
    int[] arr1thing = new int[arr1.length];
    int[] arr2thing = new int[arr1.length];

    for (int i = 0; i < arr1.length; i++) {
        arr1thing[arr1[i]] = 1;
    }
    for (int i = 0; i < arr1.length; i++) {
        arr1thing[arr2[i]] = 1;
    }

    for (int i = 0; i < arr1.length; i++) {
        if (arr1thing[i] != arr2thing[i]) {
            return false;
        }
    }
    return true;
}
